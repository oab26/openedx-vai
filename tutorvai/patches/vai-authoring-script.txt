  function applyVaiLogo() {
    var imgs = document.querySelectorAll('img.logo, header a img, header a[href*="/home"] img');
    imgs.forEach(function(logo) {
      if (logo.src.indexOf('data:image/svg') === -1) {
        logo.src = vaiDarkLogo;
        logo.alt = 'VAI';
        logo.style.maxHeight = '28px';
      }
    });

  // ─── Course Approval System ───
  var VAI_IS_ADMIN = null;

  function getCourseKeyFromUrl() {
    var match = window.location.pathname.match(/course-v1:[^/]+/);
    return match ? match[0] : null;
  }

  function getCmsBaseUrl() {
    var origin = window.location.origin;
    if (origin.indexOf('apps.local.openedx.io') !== -1) {
      return 'http://studio.local.openedx.io:8001';
    }
    return origin.replace('apps.', 'studio.').replace(/\/authoring.*/, '');
  }

  function checkIfAdmin(callback) {
    if (VAI_IS_ADMIN !== null) return callback(VAI_IS_ADMIN);
    var courseKey = getCourseKeyFromUrl();
    if (!courseKey) return callback(false);
    var xhr = new XMLHttpRequest();
    xhr.open('GET', getCmsBaseUrl() + '/export/' + courseKey, true);
    xhr.withCredentials = true;
    xhr.onreadystatechange = function() {
      if (xhr.readyState === 4) {
        VAI_IS_ADMIN = (xhr.status !== 403);
        callback(VAI_IS_ADMIN);
      }
    };
    xhr.onerror = function() { callback(false); };
    xhr.send();
  }

  // ─── Hide restricted menu items ───
  function hideRestrictedMenuItems() {
    document.querySelectorAll('a[href*="/export"], a[href*="/settings/advanced"]').forEach(function(link) {
      link.style.display = 'none';
    });
  }

  // ─── Access Restricted banner on blocked pages ───
  function showBlockedBanner() {
    var path = window.location.pathname;
    var isBlocked = path.indexOf('/settings/advanced') !== -1 || path.indexOf('/export') !== -1;

    var existing = document.getElementById('vai-blocked-banner');
    if (!isBlocked) {
      if (existing) existing.remove();
      return;
    }
    if (existing) return;

    var target = document.querySelector('article, main, [role="main"]');
    if (!target) return;

    var banner = document.createElement('div');
    banner.id = 'vai-blocked-banner';
    banner.style.cssText = 'background:#fef2f2;border:1px solid #fecaca;border-radius:8px;padding:16px 20px;margin:20px;display:flex;align-items:center;gap:12px;';
    banner.innerHTML = '<svg width="20" height="20" viewBox="0 0 20 20" fill="#dc2626"><path fill-rule="evenodd" d="M10 18a8 8 0 100-16 8 8 0 000 16zM8.707 7.293a1 1 0 00-1.414 1.414L8.586 10l-1.293 1.293a1 1 0 101.414 1.414L10 11.414l1.293 1.293a1 1 0 001.414-1.414L11.414 10l1.293-1.293a1 1 0 00-1.414-1.414L10 8.586 8.707 7.293z" clip-rule="evenodd"/></svg>'
      + '<div><strong style="color:#991b1b;font-size:14px;">Access Restricted</strong>'
      + '<p style="color:#b91c1c;font-size:13px;margin:4px 0 0;">This section requires administrator privileges. Contact the VAI admin team if you need changes made here.</p></div>';
    target.insertBefore(banner, target.firstChild);
  }

  // ─── Get current username from header ───
  function getCurrentUsername() {
    // The authoring MFE shows username in the header button like "admin admin" or "instructor1 instructor1"
    var userBtn = document.querySelector('header button[class*="user"], header button[aria-label*="Account"]');
    if (userBtn) {
      var text = userBtn.textContent.trim().split(/\s+/)[0];
      if (text) return text.toLowerCase();
    }
    // Fallback: check for username in nav
    var navLinks = document.querySelectorAll('header button, header a');
    for (var i = 0; i < navLinks.length; i++) {
      var el = navLinks[i];
      if (el.querySelector('img[alt]') && el.textContent.trim()) {
        return el.textContent.trim().split(/\s+/).pop().toLowerCase();
      }
    }
    return '';
  }

  // ─── Check if draft was saved by current user ───
  function isDraftByCurrentUser() {
    // Look for "Draft saved on ... by {username}" text in the sidebar
    var sidebar = document.querySelector('[class*="sidebar"], aside, [class*="publish"]');
    var allText = document.body.innerText || '';
    var match = allText.match(/Draft saved on .+? by (\S+)/i);
    if (!match) return false;
    var draftAuthor = match[1].toLowerCase();
    var currentUser = getCurrentUsername();
    if (!currentUser) return false;
    return draftAuthor === currentUser;
  }

  // ─── Replace Publish button with "Notify Admin for Review" in sidebar ───
  function replacePublishButton() {
    // Find all Publish buttons
    var buttons = document.querySelectorAll('button');
    buttons.forEach(function(btn) {
      var text = btn.textContent.trim();

      // Replace Publish button with Notify Admin — only if draft was made by current user
      if (text === 'Publish') {
        // Skip if we already replaced this one
        if (btn.getAttribute('data-vai-replaced') === 'true') return;

        // If draft is NOT by current user, just hide the Publish button (don't show Notify)
        if (!isDraftByCurrentUser()) {
          btn.style.display = 'none';
          return;
        }

        var courseKey = getCourseKeyFromUrl();
        if (!courseKey) return;

        // Create replacement button
        var notifyBtn = document.createElement('button');
        notifyBtn.textContent = 'Notify Admin for Review';
        notifyBtn.setAttribute('data-vai-replaced', 'true');
        // Match the original Publish button style: outlined, full-width, same padding/radius
        notifyBtn.className = btn.className;
        notifyBtn.style.cssText = 'width:100%;padding:7px 12px;background:transparent;color:rgb(10,48,85);border:1px solid rgb(10,48,85);border-radius:4px;font-size:14px;font-weight:400;cursor:pointer;transition:all 0.2s;box-sizing:border-box;';
        notifyBtn.onmouseenter = function() { if (!notifyBtn.disabled) { notifyBtn.style.background = 'rgb(10,48,85)'; notifyBtn.style.color = '#fff'; } };
        notifyBtn.onmouseleave = function() { if (!notifyBtn.disabled) { notifyBtn.style.background = 'transparent'; notifyBtn.style.color = 'rgb(10,48,85)'; } };

        notifyBtn.onclick = function() {
          notifyBtn.disabled = true;
          notifyBtn.textContent = 'Notifying...';
          notifyBtn.style.background = 'rgb(10,48,85)';
          notifyBtn.style.color = '#fff';
          notifyBtn.style.cursor = 'wait';

          var xhr = new XMLHttpRequest();
          xhr.open('POST', getCmsBaseUrl() + '/api/course-approval/notify/' + courseKey + '/', true);
          xhr.withCredentials = true;
          xhr.setRequestHeader('Content-Type', 'application/json');
          xhr.onreadystatechange = function() {
            if (xhr.readyState === 4) {
              if (xhr.status === 200) {
                notifyBtn.textContent = 'Admin Notified \u2713';
                notifyBtn.style.background = '#059669';
                notifyBtn.style.borderColor = '#059669';
                notifyBtn.style.color = '#fff';
                notifyBtn.style.cursor = 'default';
              } else {
                notifyBtn.textContent = 'Error \u2014 Try Again';
                notifyBtn.style.background = 'transparent';
                notifyBtn.style.borderColor = '#dc2626';
                notifyBtn.style.color = '#dc2626';
                notifyBtn.disabled = false;
                notifyBtn.style.cursor = 'pointer';
              }
            }
          };
          xhr.onerror = function() {
            notifyBtn.textContent = 'Error \u2014 Try Again';
            notifyBtn.style.background = 'transparent';
            notifyBtn.style.borderColor = '#dc2626';
            notifyBtn.style.color = '#dc2626';
            notifyBtn.disabled = false;
            notifyBtn.style.cursor = 'pointer';
          };
          xhr.send(JSON.stringify({}));
        };

        // Replace the Publish button
        btn.parentNode.replaceChild(notifyBtn, btn);
      }

      // Hide Discard changes
      if (text === 'Discard changes') {
        btn.style.display = 'none';
      }
    });

    // Also hide "Discard changes" links (sometimes rendered as <a> not <button>)
    document.querySelectorAll('a').forEach(function(link) {
      if (link.textContent.trim() === 'Discard changes') {
        link.style.display = 'none';
      }
    });

    // Hide Publish options inside three-dot dropdown menus on the outline page
    document.querySelectorAll('[role="menu"] button, [role="menuitem"]').forEach(function(item) {
      var itemText = item.textContent.trim();
      if (itemText === 'Publish') {
        item.style.display = 'none';
      }
    });
  }

  // ─── Remove old top banner if it exists from previous version ───
  function removeOldBanner() {
    var old = document.getElementById('vai-notify-container');
    if (old) old.remove();
  }

  // ─── Main Logic ───
  function applyApprovalUI() {
    var courseKey = getCourseKeyFromUrl();
    if (!courseKey) return;
    if (window.location.pathname.indexOf('/authoring/course/') === -1) return;

    checkIfAdmin(function(isAdmin) {
      if (isAdmin) return;
      hideRestrictedMenuItems();
      showBlockedBanner();
      removeOldBanner();
      replacePublishButton();
    });
  }

  // ─── Observer ───
  var observer = new MutationObserver(function() {
    applyVaiLogo();
    applyApprovalUI();
  });

  function startObserving() {
    if (document.body) {
      observer.observe(document.body, { childList: true, subtree: true });
      applyVaiLogo();
      applyApprovalUI();
    }
  }

  if (document.body) startObserving();
  else document.addEventListener('DOMContentLoaded', startObserving);
})();
